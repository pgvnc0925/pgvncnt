### Cluster di maturità

```
IF scoreLivello <= 15:
  maturity = "Novice"
```

### Domini primari e secondari
```
primaryDomain = argmax(scoreDominio)
// Dominio con punteggio più alto

// Se differenza tra max e 2nd max <= 3:
secondaryDomain = 2nd argmax(scoreDominio)
// Altrimenti, secondaryDomain = null
```

### Interessi primari
```

// Interesse con punteggio più alto
```

## Implementazione TypeScript (Pseudocode)

```typescript
interface ScoringRule {
  liv: number;
  acq: number;
  conv: number;
  exp: number;
  sist: number;
  strat: number;
  sistInt: number;
  psy: number;
  dati: number;
}

const scoringMatrix: Record<string, Record<number, ScoringRule>> = {
  d1: {
    1: { liv: 1, acq: 2, conv: 1, exp: 1, sist: 1, strat: 2, sistInt: 1, psy: 1, dati: 1 },
    2: { liv: 2, acq: 2, conv: 2, exp: 1, sist: 1, strat: 2, sistInt: 1, psy: 1, dati: 1 },
    // ... etc
  },
  d7: {
    0: { liv: 2, acq: 1, conv: 2, exp: 2, sist: 1, strat: 2, sistInt: 1, psy: 1, dati: 2 },
    // ... etc
  }
};

function calculateScores(answers: Record<string, number | number[]>): ScoreResult {
  let livTotal = 0, acqTotal = 0, convTotal = 0, expTotal = 0, sistTotal = 0;
  let stratTotal = 0, sistIntTotal = 0, psyTotal = 0, datiTotal = 0;
  
  for (const [question, answer] of Object.entries(answers)) {
    const rule = scoringMatrix[question][answer];
    if (!rule) continue;
    
    livTotal += rule.liv;
    acqTotal += rule.acq;
    convTotal += rule.conv;
    expTotal += rule.exp;
    sistTotal += rule.sist;
    stratTotal += rule.strat;
    sistIntTotal += rule.sistInt;
    psyTotal += rule.psy;
    datiTotal += rule.dati;
  }
  
  // Calcola maturity
  const maturity = livTotal <= 15 ? "Novice" : livTotal <= 35 ? "Practitioner" : "Advanced";
  
  // Calcola domini
  const dominiValues = { acquisizione: acqTotal, conversione: convTotal, esperienza: expTotal, sistemi: sistTotal };
  const sortedDomini = Object.entries(dominiValues).sort((a, b) => b[1] - a[1]);
  const primaryDomain = sortedDomini[0][0];
  const secondaryDomain = sortedDomini[1][1] - sortedDomini[0][1] <= -3 ? sortedDomini[1][0] : null;
  
  // Calcola interessi
  const interessiValues = { strategia: stratTotal, sistemi: sistIntTotal, psicologia: psyTotal, dati: datiTotal };
  const primaryInterest = Object.entries(interessiValues).sort((a, b) => b[1] - a[1])[0][0];
  
  return {
    scoreLivello: livTotal,
    scoreDomini: dominiValues,
    scoreInteressi: interessiValues,
    maturity,
    primaryDomain,
    secondaryDomain,
    primaryInterest,
  };
}
```
